#!/bin/bash
# monitor.sh — Monitoreo de espacio, inodos y archivos del servidor

UMBRAL_LOGS=20     # alerta si hay más de N archivos de log

else
    echo "OK: $LOGS archivos .log"
fi


echo '======================================='
EOF
